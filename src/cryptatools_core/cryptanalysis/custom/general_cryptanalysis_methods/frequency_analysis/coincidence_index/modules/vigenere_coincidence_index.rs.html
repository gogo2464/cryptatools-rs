22    /// Generate coincidence index of a specific key_size choosen poorly for a specified plain text
23    /// 
24    /// This function does not only aim to be used against cipher text.
25    /// The goal of this function is also to provide statistics about plain text coincidence index in order to decipher another encrypted text.
26    /// 
27    /// ```
28    /// use cryptatools_core::utils::{convert::Encode, alphabets::Alphabet};
29    /// use cryptatools_core::cryptanalysis::custom::general_cryptanalysis_methods::frequency_analysis::coincidence_index::modules::vigenere_coincidence_index::VigenereCoincidenceIndexGenerator;
30    /// 
31    /// use approx::assert_abs_diff_eq;
32    /// 
33    /// let alphabet = Alphabet::new_empty().ascii_printable_only_encoding();
34    /// 
35    /// let mut vcig = VigenereCoincidenceIndexGenerator::new(alphabet.into());
36    /// let mut ci = vcig.generate_coincidence_index_for_key(0, Encode::from_ascii_to_u8(String::from("The ennemy will never attack!")));
37    /// assert_abs_diff_eq!(ci, 0.0541871921182266, epsilon = 1e-16);
38    /// ci = vcig.generate_coincidence_index_for_key(5, Encode::from_ascii_to_u8(String::from("The ennemy will never attack! I think the ennemy is simply too coward in order to lead an attack. He will hesistate too much. Prepare your troops for a very very long defense of this siege.")));
39    /// assert_abs_diff_eq!(ci, 0.02, epsilon = 1e-1);
40    /// ci = vcig.generate_coincidence_index_for_key(10, Encode::from_ascii_to_u8(String::from("The ennemy will never attack!")));
41    /// assert_abs_diff_eq!(ci, 0.01, epsilon = 1e-1);
42    /// 
43    /// ci = vcig.generate_coincidence_index_for_key(10, Encode::from_ascii_to_u8(String::from("The ennemy will never attack!")));
44    /// assert_abs_diff_eq!(ci, 0.01, epsilon = 1e-1);
45    /// ```
46    pub fn generate_coincidence_index_for_key(&self, key_size: usize, input: Vec<u8>) -> f64 {
47        let mut rng = rand::thread_rng();
48        let mut key = vec![];
49        for _i in 0..key_size {
50            let random_value = rng.gen_range(0..self.alphabet.encoding.len());
51            let random_byte = self.alphabet.encoding.right_values().nth(random_value).unwrap()[0];// TODO get multichars // .sorted() ?
52            key.push(vec![random_byte]);
53        }
54
55        let vig: VigenereNoTable = VigenereNoTable::new(self.alphabet.clone());
56        let vigenere_coincidence_index_guesser = CoincidenceIndexGuesser::new(self.alphabet.clone());
57        let cipher_text = vig.encrypt(input, key);
58        let coincidence_index = vigenere_coincidence_index_guesser.guess_coincidence_index(cipher_text);
59
60        coincidence_index
61    }
62
63    /// Generate coincidence index corresponding for a sepcific size coincidence index values of an external file in plain text.
64    /// 
65    /// 
66    /// The coincidence index depends of the alphabet.
67    /// ```
68    /// use once_cell::sync::Lazy;
69    /// use cryptatools_core::utils::{convert::Encode,  alphabets::Alphabet};
70    /// use cryptatools_core::cryptanalysis::custom::general_cryptanalysis_methods::frequency_analysis::coincidence_index::modules::vigenere_coincidence_index::VigenereCoincidenceIndexGenerator;
71    /// use approx::assert_abs_diff_eq;
72    /// 
73    /// 
74    /// let alphabet = Alphabet::new_empty().ascii_printable_only_encoding();
75    /// let mut vcig = VigenereCoincidenceIndexGenerator::new(alphabet.into());
76    /// let mut ci = vcig.generate_coincidence_index_for_key_from_file(5, String::from("./data/text-corpus-for-statistics/gutenberg/austen-emma.txt"));
77    /// 
78    /// assert_abs_diff_eq!(ci, 0.01, epsilon = 1e-1);
79    /// ```
80    pub fn generate_coincidence_index_for_key_from_file(&self, key_size: usize, plain_text_ascii_file_input: String) -> f64 {
81        let file_path = Path::new(&plain_text_ascii_file_input);
82        let file_content = match fs::read_to_string(file_path) {
83            Ok(file) => file.replace("\r\n", "\n"),
84            Err(error) => panic!("{0}", error),
85        };
86
87        let encoded_file_content = Encode::encode(&self.alphabet, file_content);
88        let coincidence_index_found = self.generate_coincidence_index_for_key(key_size, encoded_file_content);
89        coincidence_index_found
90    }
91}
