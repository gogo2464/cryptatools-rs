20    /// Guess coincidence index of `cipher_text_input`.
21    /// 
22    /// ```
23    /// use cryptatools_core::cryptanalysis::custom::general_cryptanalysis_methods::frequency_analysis::coincidence_index::engine::coincidence_index::CoincidenceIndexGuesser;
24    /// use cryptatools_core::utils::alphabets::Alphabet;
25    /// use cryptatools_core::utils::convert::Encode;
26    /// 
27    /// use approx::assert_abs_diff_eq;
28    /// 
29    /// let alphabet = Alphabet::new_empty().ascii_printable_only_encoding();
30    /// 
31    /// let plain_text = String::from("Hello! How are you? I am fine and you?");
32    /// let pseudo_cipher_text = Encode::from_ascii_to_u8(plain_text);
33    /// let c = CoincidenceIndexGuesser::new(alphabet.into());
34    /// let coincidence_index: f64 = c.guess_coincidence_index(pseudo_cipher_text);
35    /// assert_abs_diff_eq!(coincidence_index, 0.06543385490753911, epsilon = 1e-16);
36    /// ```
37    pub fn guess_coincidence_index(&self, cipher_text_input: Vec<u8>) -> f64 {
38        let cipher_text_size: f64 = cipher_text_input.len() as f64;
39        if cipher_text_size <= 200.0 {
40            println!("Warning: The cipher text input is {0} bytes. Probably {0} characters. It may be too short. You should provide more input characters.", cipher_text_size);
41        }
42
43        let mut coincidence_index: f64 = 0.0;
44        for u8_byte_alphabet in self.alphabet.encoding.right_values() {//TODO: convert to f64
45            let apparition_count: f64 = cipher_text_input.iter().filter(|&n| *n == u8_byte_alphabet[0]).count() as f64;// the [0] is a quick hack to avoid to find an algorithm to compare a set of bytes with some bytes of different size.
46            let sum_apparition_alphabet: f64 = apparition_count * (apparition_count - 1.0);
47            let divide_characters: f64 = cipher_text_size * (cipher_text_size - 1.0);
48            let result: f64 = sum_apparition_alphabet / divide_characters;
49            coincidence_index += result;
50        }
51
52        coincidence_index
53    }

55    /// Guess coincidence index of an unencrypted plain text external file.
56    /// 
57    /// This function does not only aim to be used against cipher text.
58    /// The goal of this function is also to provide statistics about plain text coincidence index in order to decipher another encrypted text.
59    /// ```
60    /// use once_cell::sync::Lazy;
61    /// use cryptatools_core::utils::alphabets::Alphabet;
62    ///  use cryptatools_core::cryptanalysis::custom::general_cryptanalysis_methods::frequency_analysis::coincidence_index::engine::coincidence_index::CoincidenceIndexGuesser;
63    /// use approx::assert_abs_diff_eq;
64    /// 
65    /// let alphabet = Alphabet::new_empty().ascii_printable_only_encoding();
66    /// 
67    /// let path = String::from("./data/text-corpus-for-statistics/gutenberg/austen-emma.txt");
68    /// let mut c = CoincidenceIndexGuesser::new(alphabet.into());
69    /// assert_abs_diff_eq!(c.guess_coincidence_index_statistics_from_file(path.clone()), 0.06742905478018858, epsilon = 1e-15);
70    /// ```
71    pub fn guess_coincidence_index_statistics_from_file(&self, file_name: String) -> f64 {
72        let file_path = Path::new(&file_name);
73        let file_content = match fs::read_to_string(file_path) {
74            Ok(file) => file,
75            Err(error) => panic!("{0}", error),
76        };
77
78        let bytes_file_content = Encode::encode(&self.alphabet, file_content);
79        let coincidence_index = self.guess_coincidence_index(bytes_file_content);
80        
81        coincidence_index
82    }
