21    /// Guess coincidence index of `cipher_text_input`.
22    /// 
23    /// ```
24    /// use cryptatools_core::cryptanalysis::custom::general_cryptanalysis_methods::frequency_analysis::coincidence_index::CoincidenceIndexGuesser;
25    /// use cryptatools_core::utils::alphabets::Alphabet;
26    /// use cryptatools_core::utils::convert::Encode;
27    /// 
28    /// use approx::assert_abs_diff_eq;
29    /// 
30    /// let alphabet = Alphabet::new_empty().ascii_printable_only_encoding();
31    /// 
32    /// let plain_text = String::from("Hello! How are you? I am fine and you?");
33    /// let pseudo_cipher_text = Encode::from_ascii_to_u8(plain_text);
34    /// let c = CoincidenceIndexGuesser::new(alphabet.into());
35    /// let coincidence_index: f64 = c.guess_coincidence_index(pseudo_cipher_text);
36    /// assert_abs_diff_eq!(coincidence_index, 0.06543385490753911, epsilon = 1e-16);
37    /// ```
38    pub fn guess_coincidence_index(&self, cipher_text_input: Vec<u8>) -> f64 {
39        let cipher_text_size: f64 = cipher_text_input.len() as f64;
40        if cipher_text_size <= 200.0 {
41            println!("Warning: The cipher text input is {0} bytes. Probably {0} characters. It may be too short. You should provide more input characters.", cipher_text_size);
42        }
43
44        let mut coincidence_index: f64 = 0.0;
45        for u8_byte_alphabet in self.alphabet.encoding.right_values() {//TODO: convert to f64
46            let apparition_count: f64 = cipher_text_input.iter().filter(|&n| *n == u8_byte_alphabet[0]).count() as f64;// the [0] is a quick hack to avoid to find an algorithm to compare a set of bytes with some bytes of different size.
47            let sum_apparition_alphabet: f64 = apparition_count * (apparition_count - 1.0);
48            let divide_characters: f64 = cipher_text_size * (cipher_text_size - 1.0);
49            let result: f64 = sum_apparition_alphabet / divide_characters;
50            coincidence_index += result;
51        }
52
53        coincidence_index
54    }

56    /// Guess coincidence index of an unencrypted plain text external file.
57    /// 
58    /// This function does not only aim to be used against cipher text.
59    /// The goal of this function is also to provide statistics about plain text coincidence index in order to decipher another encrypted text.
60    /// ```
61    /// use once_cell::sync::Lazy;
62    /// use cryptatools_core::utils::alphabets::Alphabet;
63    /// use cryptatools_core::cryptanalysis::custom::general_cryptanalysis_methods::frequency_analysis::coincidence_index::CoincidenceIndexGuesser;
64    /// use approx::assert_abs_diff_eq;
65    /// 
66    /// let alphabet = Alphabet::new_empty().ascii_printable_only_encoding();
67    /// 
68    /// let path = String::from("./data/text-corpus-for-statistics/gutenberg/austen-emma.txt");
69    /// let mut c = CoincidenceIndexGuesser::new(alphabet.into());
70    /// assert_abs_diff_eq!(c.guess_coincidence_index_statistics_from_file(path.clone()), 0.06742905478018858, epsilon = 1e-15);
71    /// ```
72    pub fn guess_coincidence_index_statistics_from_file(&self, file_name: String) -> f64 {
73        let file_path = Path::new(&file_name);
74        let file_content = match fs::read_to_string(file_path) {
75            Ok(file) => file,
76            Err(error) => panic!("{0}", error),
77        };
78
79        let bytes_file_content = Encode::encode(&self.alphabet, file_content);
80        let coincidence_index = self.guess_coincidence_index(bytes_file_content);
81        
82        coincidence_index
83    }

90impl CoincidenceIndexGenerator {
91    pub fn new(alphabet: Arc<Alphabet>) -> Self {
92        CoincidenceIndexGenerator {
93            alphabet: alphabet,
94        }
95    }
96    /// Generate coincidence index of a specific key_size choosen poorly for a specified plain text
97    /// 
98    /// This function does not only aim to be used against cipher text.
99    /// The goal of this function is also to provide statistics about plain text coincidence index in order to decipher another encrypted text.
100    /// 
101    /// ```
102    /// use cryptatools_core::utils::{convert::Encode, alphabets::Alphabet};
103    /// use cryptatools_core::cryptanalysis::custom::general_cryptanalysis_methods::frequency_analysis::coincidence_index::CoincidenceIndexGenerator;
104    /// 
105    /// use approx::assert_abs_diff_eq;
106    /// 
107    /// let alphabet = Alphabet::new_empty().ascii_printable_only_encoding();
108    /// 
109    /// let mut vcig = CoincidenceIndexGenerator::new(alphabet.into());
110    /// let mut ci = vcig.generate_coincidence_index_for_key(0, Encode::from_ascii_to_u8(String::from("The ennemy will never attack!")));
111    /// assert_abs_diff_eq!(ci, 0.0541871921182266, epsilon = 1e-16);
112    /// ci = vcig.generate_coincidence_index_for_key(5, Encode::from_ascii_to_u8(String::from("The ennemy will never attack! I think the ennemy is simply too coward in order to lead an attack. He will hesistate too much. Prepare your troops for a very very long defense of this siege.")));
113    /// assert_abs_diff_eq!(ci, 0.02, epsilon = 1e-1);
114    /// ci = vcig.generate_coincidence_index_for_key(10, Encode::from_ascii_to_u8(String::from("The ennemy will never attack!")));
115    /// assert_abs_diff_eq!(ci, 0.01, epsilon = 1e-1);
116    /// 
117    /// ci = vcig.generate_coincidence_index_for_key(10, Encode::from_ascii_to_u8(String::from("The ennemy will never attack!")));
118    /// assert_abs_diff_eq!(ci, 0.01, epsilon = 1e-1);
119    /// ```
120    pub fn generate_coincidence_index_for_key(&self, key_size: usize, input: Vec<u8>) -> f64 {
121        let mut rng = rand::thread_rng();
122        let mut key = vec![];
123        for _i in 0..key_size {
124            let random_value = rng.gen_range(0..self.alphabet.encoding.len());
125            let random_byte = self.alphabet.encoding.right_values().nth(random_value).unwrap()[0];// TODO get multichars // .sorted() ?
126            key.push(vec![random_byte]);
127        }
128
129        let vig: VigenereNoTable = VigenereNoTable::new(self.alphabet.clone());
130        let vigenere_coincidence_index_guesser = CoincidenceIndexGuesser::new(self.alphabet.clone());
131        let cipher_text = vig.encrypt(input, key);
132        let coincidence_index = vigenere_coincidence_index_guesser.guess_coincidence_index(cipher_text);
133
134        coincidence_index
135    }

137    /// Generate coincidence index corresponding for a sepcific size coincidence index values of an external file in plain text.
138    /// 
139    /// 
140    /// The coincidence index depends of the alphabet.
141    /// ```
142    /// use once_cell::sync::Lazy;
143    /// use cryptatools_core::utils::{convert::Encode,  alphabets::Alphabet};
144    /// use cryptatools_core::cryptanalysis::custom::general_cryptanalysis_methods::frequency_analysis::coincidence_index::CoincidenceIndexGenerator;
145    /// use approx::assert_abs_diff_eq;
146    /// 
147    /// 
148    /// let alphabet = Alphabet::new_empty().ascii_printable_only_encoding();
149    /// let mut vcig = CoincidenceIndexGenerator::new(alphabet.into());
150    /// let mut ci = vcig.generate_coincidence_index_for_key_from_file(5, String::from("./data/text-corpus-for-statistics/gutenberg/austen-emma.txt"));
151    /// 
152    /// assert_abs_diff_eq!(ci, 0.01, epsilon = 1e-1);
153    /// ```
154    pub fn generate_coincidence_index_for_key_from_file(&self, key_size: usize, plain_text_ascii_file_input: String) -> f64 {
155        let file_path = Path::new(&plain_text_ascii_file_input);
156        let file_content = match fs::read_to_string(file_path) {
157            Ok(file) => file.replace("\r\n", "\n"),
158            Err(error) => panic!("{0}", error),
159        };
160
161        let encoded_file_content = Encode::encode(&self.alphabet, file_content);
162        let coincidence_index_found = self.generate_coincidence_index_for_key(key_size, encoded_file_content);
163        coincidence_index_found
164    }
165}
